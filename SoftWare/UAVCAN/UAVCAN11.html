本文介绍了如何利用candump和gawk工具快速创建一个Cyphal协议解析的小工具，以替代安装复杂的yakut进行调试。通过解析7D55心跳广播id的单帧消息，展示了解析过程的关键步骤。
摘要由CSDN通过智能技术生成

经常需要调试Cyphal协议，yakut的安装比较麻烦，所以就用candump和gawk实现了一个协议解析工具。

这个candump导出的数据，很明显这是一条心跳广播，7D55是心跳广播的id，而且还是条单帧消息。

can1 107D552A [8] 00 00 00 00 00 01 A1 E0
can1 107D552A [8] 01 00 00 00 00 01 A1 E1
can1 107D552A [8] 02 00 00 00 00 01 A1 E2
can1 107D552A [8] 03 00 00 00 00 01 A1 E3

贴上我们的awk脚本：

usage_and_exit () {
  echo -e "\e[32m$0\e[0m support option is :
  \e[32m-v|--version\e[0m show the program version
  \e[32m-d|--debug\e[0m open debug mode
  \e[32m-m|--monitor\e[0m print online node and do message statistic
  \e[32m-i|--interface\e[0m specify the can interface to listen
  \e[32m-c|--cycle\e[0m set frequency calculate cycle, uint is second.
  \e[32m-e|--print_error\e[0m print toggle error and transfer id not continue error.
  \e[32m-h|--help\e[0m show this help info as you see"
  exit 1
}

POSITIONAL=()
open_monitor=1
interface="can0"
open_debug=0
print_error=0;
cycle=1.0;

while (( $# > 0 )); do
  case "${1}" in
    -h|--help)
    usage_and_exit
    shift # shift once since flags have no values
    ;;
    -v|--version)
    version
    shift # shift once since flags have no values
    ;;
    -e|--print_error)
    print_error=1;
    open_monitor=0;
    shift # shift once since flags have no values
    ;;
    -d|--debug)
    open_debug=1
    open_monitor=0
    shift # shift once since flags have no values
    ;;
    -m|--monitor)
    open_monitor=1
    open_debug=0
    shift # shift once since flags have no values
    ;;
    -c|--cycle)
    numOfArgs=1 # number of switch arguments
    if (( $# < numOfArgs + 1 )); then
      shift $#
    else
      cycle=${2}
      shift $((numOfArgs + 1)) # shift 'numOfArgs + 1' to bypass switch and its value
    fi
    ;;
    -i|--interface)
    numOfArgs=1 # number of switch arguments
    if (( $# < numOfArgs + 1 )); then
      shift $#
    else
      interface=${2}
      shift $((numOfArgs + 1)) # shift 'numOfArgs + 1' to bypass switch and its value
    fi
    ;;
    *) # unknown flag/switch
    POSITIONAL+=("${1}")
    shift
    ;;
  esac
  done

  set -- "${POSITIONAL[@]}" # restore positional params

  candump -ta ${interface} | gawk -v open_debug=${open_debug} -v open_monitor=${open_monitor} -v cycle=${cycle} -v print_error=${print_error} 'BEGIN {

  r_id_name[100] = "SetVelocity";
  r_id_name[104] = "SetMode";
  r_id_name[103] = "RequestParameter";
  r_id_name[1] = "NodeManage";
  r_id_name[251] = "WriteFlash";
  r_id_name[252] = "StopOTA";
  r_id_name[249] = "StartOTA";
  r_id_name[248] = "WriteUpdatee";
  r_id_name[32] = "light";
  r_id_name[33] = "beeper";
  r_id_name[250] = "UpdateeInterate";
  r_id_name[2] = "BoardInfo";

  id_name[7509] = "heart_beat";
  id_name[181] = "battery_status";
  id_name[1009] = "ir";
  id_name[1011] = "power_monitor";
  id_name[2] = "GlobalHealth";
  id_name[1] = "Emergency";
  id_name[1002] = "sonar";
  id_name[1003] = "imu";
  id_name[1100] = "odometry";
  id_name[1000] = "lidar";
  id_name[1101] = "motor_status";
  id_name[1005] = "power_button";
  id_name[7168] = "time_sync";
  # add your id and name map here

  can_id_index = 2;

  getline
  #line example: (001.000) can0 11111 [8] 55555
  if($1 ~/^\(/)
    can_id_index = 3;

  while (getline) {

  # have timestamp
  if(can_id_index == 3) {
    timestamp = strtonum(substr($1, 2));
  }

  #line example: can0 1122233 [8] 5555555
  can_id = strtonum("0x"$can_id_index);

  # not can 2.0 B
  if(can_id < 2048) {
    continue;
  }
  source_id = and(can_id, 0x0000007F);

  # 0 message
  # 1 response
  # 2 request
  type = and(rshift(can_id, 24), 0x03);

  priority = and(rshift(can_id, 26), 0x07);

  tail_byte = strtonum("0x"$NF);
  start_bit = rshift(tail_byte, 7);
  end_bit = and(rshift(tail_byte, 6), 0x01);
  toggle_bit = and(rshift(tail_byte, 5), 0x01);
  transfer_id = and(tail_byte, 31);

  if(type == 0) {
    subject_id = and(rshift(can_id, 8), 0x1FFF);
    if(subject_id == 7509) {
          heart_beat[source_id]++
    }
  } else if(open_debug && (type == 2 || type == 3)) {
    dest_id = and(rshift(can_id, 7), 0x7F);
    subject_id = and(rshift(can_id, 14), 0x1FF);

    if(start_bit) {
      frame_count[subject_id][type] = 0;
    }
    frame = frame_count[subject_id][type]
    message[subject_id][type][frame] = $0;
    frame_count[subject_id][type]++;
    
    if(end_bit) {
      if(type == 2)
        print "response: ";
      else 
        print "request: ";
      print "\tname: " r_id_name[subject_id];
      print "\tsource id: " source_id;
      print "\tdest_id: " dest_id;
      print "\tsubject_id: " subject_id;
      print "\ttransfer_id: " transfer_id;

      for(i = 0; i < frame_count[subject_id][type]; i++) {
        printf("\tdata[%d]: %s\n", i, message[subject_id][type][i]);
      }
      frame_count[subject_id][type] = 0;
    }
  }

  session_id[source_id][subject_id] = transfer_id;

  if(start_bit) {
    toggle_error = 0;
    start_of_transfer[subject_id] = start_bit;
  }
  if(end_bit) {
    end_of_transfer[subject_id] = end_bit;
  }

  if(!start_bit) {
    # check toggle
    if(last_toggle_bit[subject_id] == toggle_bit) {
      toggle_error = 1;
      toggle_er[type][subject_id]++;
      if(open_error) {
	      print "toggle error, frame may be lost: " $0;
	      print "subject_id: " subject_id;
      }
    }
  }
  last_toggle_bit[subject_id] = toggle_bit;

  # receive one complete message
  if(!toggle_error && start_of_transfer[subject_id] && end_of_transfer[subject_id]) {
    count[type][subject_id]++;
    start_of_transfer[subject_id] = 0;
    end_of_transfer[subject_id] = 0;
  }

  #judge session id is continue or not
  if(end_bit) {
  if(print_error && (last_session_id[source_id][subject_id] + 1) % 32 != session_id[source_id][subject_id]) {
     printf("subject_id: %04d, source_id: %03d, transfer id not continue, last: %02d, now: %02d\n", subject_id, source_id, last_session_id[source_id][subject_id], session_id[source_id][subject_id]);
  }
     last_session_id[source_id][subject_id] = session_id[source_id][subject_id];
  }

  # statics, print online node, message fequency, 1 second update
  if(open_monitor && timestamp) {
    if(timestamp - last_timestamp > cycle) {
            printf("\033cversion: 0.2, opencyphal debug tool by hongjh\n");
            printf("write at 2022/08/08 02:05\n")
            printf("\n----------------------------------------------------------------------------------\n")

            for(i = 0; i < 128; i++) {
                    if(heart_beat[i]) {
                      printf("\tnode: %04d online\n", i);
                      heart_beat[i] = 0;
                    }
            }
            printf("-------------------------------------request--------------------------------------\n\n")

            for(i = 0; i < 512; i++) { 
                    if(count[3][i]) {
                      printf("request name: %12s, subject id: %04d, frequency: %04d\n",r_id_name[i], i, count[2][i]);
                      count[3][i] = 0;
		      toogle_er[3][subject_id] = 0;
		    }
	    }
            printf("-------------------------------------publish--------------------------------------\n\n")
            for(i = 0; i < 8192; i++) { 
                    if(count[0][i]) {
                      printf("message name: %14s, subject id: %04d, frequency: %04d, toggle_error: %02d\n",id_name[i], i, count[0][i], toggle_er[subject]);
                      count[0][i] = 0;
		      toogle_er[0][subject_id] = 0;
                    }
            }
            printf("-----------------------------------------------------------------------------------\n")
            last_timestamp = timestamp;
    }
  }
  }
}'
